function showContent() {
    var dropDown = document.getElementById("dropContent");

    if (dropDown.style.display === 'block') {
        dropDown.style.display = 'none';
    }
    else {
        dropDown.style.display = 'block';
    }
}

function hideContent() {
    var dropDown = document.getElementById("dropContent");
    dropDown.style.display = "none";
}
